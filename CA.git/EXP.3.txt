#include <stdio.h>
#include <time.h>
int main() {
    int arr[100], n, i, j, temp;
    int swapped;
    clock_t start, end;
    double cpu_time_used;
    printf("Enter number of elements (max 100): ");
    scanf("%d", &n);
    if (n <= 0 || n > 100) {
        printf("Invalid number of elements.\n");
        return 1;
    }
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    // Start time
    start = clock();
    // Optimized Bubble Sort
    for (i = 0; i < n - 1; i++) {
        swapped = 0;
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;
            }
        }
        if (swapped == 0)
            break;  // Stop if already sorted
    }
    // End time
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\nSorted array in ascending order:\n");
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n\nTime taken: %.10f seconds\n", cpu_time_used);
    printf("Time Complexity (Worst & Average): O(n^2)\n");
    printf("Time Complexity (Best Case): O(n)\n");
    printf("Space Complexity: O(1)\n");
    printf("\n-----------------------------------\n");
    printf("Name : Vinayak Patil\n");
    printf("PRN  : 22458010261\n");
    return 0;
