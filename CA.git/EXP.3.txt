#include <stdio.h>
#include <time.h>
int main() {
    int a[100], n, i, j, temp, swapped;
    clock_t start = clock();
    printf("Enter number of elements (max 100): ");
    scanf("%d", &n);
    if (n < 1 || n > 100) return 1;
    printf("Enter elements:\n");
    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);
    // Bubble Sort
    for (i = 0; i < n - 1; i++) {
        swapped = 0;
        for (j = 0; j < n - i - 1; j++) {
            if (a[j] > a[j + 1]) {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
                swapped = 1;
            }
        }
        if (!swapped) break;
    }
    clock_t end = clock();
    printf("\nSorted array:\n");
    for (i = 0; i < n; i++)
        printf("%d ", a[i]);
    printf("\nTime: %.6f sec\n", (double)(end - start) / CLOCKS_PER_SEC);
    printf("Time Complexity: O(n^2), Best: O(n), Space: O(1)\n");
}
