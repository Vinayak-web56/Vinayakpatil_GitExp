#include <stdio.h>
#include <string.h>
#include <time.h>
#define MAX 100
int max(int a, int b) {
    return (a > b) ? a : b;
}
void LCS(char X[], char Y[]) {
    int m = strlen(X);
    int n = strlen(Y);
    int dp[MAX][MAX];
    // Build DP table
    for (int i = 0; i <= m; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == 0 || j == 0)
                dp[i][j] = 0;
            else if (X[i-1] == Y[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    printf("\nLength of LCS = %d\n", dp[m][n]);
    // Backtrack to find LCS string
    int index = dp[m][n];
    char lcs[index + 1];
    lcs[index] = '\0';
    int i = m, j = n;
    while (i > 0 && j > 0) {
        if (X[i-1] == Y[j-1]) {
            lcs[index - 1] = X[i-1];
            i--;
            j--;
            index--;
        }
        else if (dp[i-1][j] > dp[i][j-1])
            i--;
        else
            j--;
    }
    printf("LCS String = %s\n", lcs);
}
int main() {
    char X[MAX], Y[MAX];
    clock_t start, end;
    double cpu_time_used;
    printf("Enter first string: ");
    scanf("%s", X);
    printf("Enter second string: ");
    scanf("%s", Y);
    start = clock();
    LCS(X, Y);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("Execution Time: %.10f seconds\n", cpu_time_used);
    printf("\nTime Complexity: O(m*n)\n");
    printf("Space Complexity: O(m*n)\n");
    printf("\n-----------------------------------\n");
    printf("Name : Vinayak Patil\n");
    printf("PRN  : 22458010261\n");
    return 0;
}
