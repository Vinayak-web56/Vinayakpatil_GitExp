#include <stdio.h>
#include <time.h>
long long factorial_iterative(int n) {
    long long result = 1;
    for (int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}
long long factorial_recursive(int n) {
    if (n <= 1)
        return 1;
    return n * factorial_recursive(n - 1);
}
int main() {
    int n;
    clock_t start, end;
    double cpu_time_used;
    long long result;
 printf("Enter a number to find factorial (0-20): ");
    scanf("%d", &n);
    if (n < 0) {
        printf("Factorial is not defined for negative numbers.\n");
        return 1;
    }
 if (n > 20) {
        printf("Warning: Result may overflow for n > 20 using long long.\n\n");
    }
    start = clock();
    result = factorial_iterative(n);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
  printf("\n[Iterative] Factorial of %d = %lld\n", n, result);
    printf("[Iterative] Time taken: %.10f seconds\n", cpu_time_used);
    printf("[Iterative] Space Complexity: O(1)\n");
    start = clock();
    result = factorial_recursive(n);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
 printf("\n[Recursive] Factorial of %d = %lld\n", n, result);
    printf("[Recursive] Time taken: %.10f seconds\n", cpu_time_used);
    printf("[Recursive] Space Complexity: O(n) (recursion stack)\n");
}
