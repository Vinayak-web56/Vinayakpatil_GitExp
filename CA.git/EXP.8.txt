#include<stdio.h>
#define MAX 10

int adj[MAX][MAX], visited[MAX], n;

void bfs(int start){
    int q[MAX], f = 0, r = 0;

    q[r++] = start;
    visited[start] = 1;

    while(f < r){
        int v = q[f++];
        printf("%d ", v);

        for(int i = 0; i < n; i++){
            if(adj[v][i] && !visited[i]){
                q[r++] = i;
                visited[i] = 1;
            }
        }
    }
}

int main(){
    int start;

    printf("Enter vertices: ");
    scanf("%d", &n);

    printf("Enter matrix:\n");
    for(int i = 0; i < n; i++)
        for(int j = 0; j < n; j++)
            scanf("%d", &adj[i][j]);

    printf("Start vertex: ");
    scanf("%d", &start);

    bfs(start);

    return 0;
}










#include<stdio.h>
#define MAX 10

int adj[MAX][MAX], visited[MAX], n;

void dfs(int v){
    printf("%d ", v);
    visited[v] = 1;

    for(int i = 0; i < n; i++){
        if(adj[v][i] && !visited[i]){
            dfs(i);
        }
    }
}

int main(){
    int start;

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter adjacency matrix:\n");
    for(int i = 0; i < n; i++)
        for(int j = 0; j < n; j++)
            scanf("%d", &adj[i][j]);

    printf("Enter starting vertex: ");
    scanf("%d", &start);

    dfs(start);

    return 0;
}
