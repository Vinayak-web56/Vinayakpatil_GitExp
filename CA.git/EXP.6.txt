#include <iostream>
using namespace std;

int main() {
    int n, W;

    cout << "Enter number of items: ";
    cin >> n;

    int val[n], wt[n];

    cout << "Enter values:\n";
    for (int i = 0; i < n; i++)
        cin >> val[i];

    cout << "Enter weights:\n";
    for (int i = 0; i < n; i++)
        cin >> wt[i];

    cout << "Enter capacity: ";
    cin >> W;

    int dp[n + 1][W + 1];

    // Fill DP table
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            if (i == 0 || w == 0)
                dp[i][w] = 0;
            else if (wt[i - 1] <= w)
                dp[i][w] = max(val[i - 1] + dp[i - 1][w - wt[i - 1]], dp[i - 1][w]);
            else
                dp[i][w] = dp[i - 1][w];
        }
    }

    // 🔥 ADDED BLOCK: PRINT DP TABLE
    cout << "\nDynamic Programming Table:\n";
    for (int i = 0; i <= n; i++) {
        for (int w = 0; w <= W; w++) {
            cout << dp[i][w] << "\t";
        }
        cout << endl;
    }

    cout << "\nMaximum value = " << dp[n][W];

    return 0;
}
