#include <stdio.h>
#include <time.h>
#define MAX_ITEMS 20
#define MAX_CAPACITY 100
// Function to return maximum of two numbers
int max(int a, int b) {
    return (a > b) ? a : b;
}
// Function to solve 0/1 Knapsack using Dynamic Programming
int knapSack(int W, int wt[], int val[], int n) {
    int i, w;
    int K[MAX_ITEMS + 1][MAX_CAPACITY + 1];
    for (i = 0; i <= n; i++) {
        for (w = 0; w <= W; w++) {
            if (i == 0 || w == 0)
                K[i][w] = 0;
            else if (wt[i - 1] <= w)
                K[i][w] = max(val[i - 1] + K[i - 1][w - wt[i - 1]],
                              K[i - 1][w]);
            else
                K[i][w] = K[i - 1][w];
        }
    }
    // Display DP Table
    printf("\nDynamic Programming Table:\n");
    for (i = 0; i <= n; i++) {
        for (w = 0; w <= W; w++) {
            printf("%3d ", K[i][w]);
        }
        printf("\n");
    }
    return K[n][W];
}
int main() {
    int val[MAX_ITEMS], wt[MAX_ITEMS];
    int W, n, i;
    clock_t start, end;
    double cpu_time_used;
    printf("Enter number of items (max %d): ", MAX_ITEMS);
    scanf("%d", &n);
    if (n <= 0 || n > MAX_ITEMS) {
        printf("Invalid number of items.\n");
        return 1;
    }
    printf("Enter values (profits) of %d items:\n", n);
    for (i = 0; i < n; i++)
        scanf("%d", &val[i]);
    printf("Enter weights of %d items:\n", n);
    for (i = 0; i < n; i++)
        scanf("%d", &wt[i]);
    printf("Enter capacity of Knapsack (max %d): ", MAX_CAPACITY);
    scanf("%d", &W);
    if (W < 0 || W > MAX_CAPACITY) {
        printf("Invalid capacity.\n");
        return 1;
    }
    start = clock();
    int maxValue = knapSack(W, wt, val, n);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\nMaximum value that can be put in the knapsack = %d\n", maxValue);
    printf("Execution Time: %.10f seconds\n", cpu_time_used);
    printf("\nTime Complexity: O(nW)\n");
    printf("Space Complexity: O(nW)\n");
    printf("\n-----------------------------------\n");
    printf("Name : Vinayak Patil\n");
    printf("PRN  : 22458010261\n");
   return 0;
}
